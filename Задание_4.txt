def add(a, b):
    return a + b

def test_add():
    # Тест 1: Проверка сложения положительных чисел
    assert add(2, 3) == 5, "Ошибка в сложении положительных чисел"
    
    # Тест 2: Проверка сложения отрицательных чисел
    assert add(-1, -1) == -2, "Ошибка в сложении отрицательных чисел"
    
    # Тест 3: Проверка сложения с нулём
    assert add(5, 0) == 5, "Ошибка в сложении с нулём"
    
    print("Все тесты пройдены успешно!")

test_add()
